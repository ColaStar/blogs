function inherit(C, P) {
  var F = function() {};
  F.prototype = P.prototype;
  C.prototype = new F();	// 临时构造函数
  
  C.prototype.constructor = C;	// 修复constructor
  C.super = P; // 存储超类
}

function Student(age, num) {
  Student.super.call(this. age);
  this.num = num;
}
